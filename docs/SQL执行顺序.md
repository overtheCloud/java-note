(8) SELECT  (9) DISTINCT <SELECT_LIST>  
(1) FROM <LEFT_TABLE>  <JOIN_TYPE>   
(2) JOIN <RIGHT_TABLE>   
(3) ON  <JOIN_CONDITION>  
(4) WHERE  <WHERE_CONDITION>  
(5) GROUP BY <GROUP BY_LIST>  
(6) WITH <CUBE|ROLLUP>  
(7) HAVING  <HAVING_CONDITION>  
(10) ORDER BY <ORDER_BY_LIST>  
(11) LIMIT <LIMIT_NUMBER>
